<!-- 1. Drop in the widget (one tag, no npm) -->
<script
  src="https://klavity.in/widget.js"
  data-project="proj_abc123"
  data-replay="on"
  defer
></script>

<!-- 2. Wire up identity + metadata (after your auth resolves) -->
<script>
authPromise.then(function(user) {
  window.Klavity?.identify({
    id:    user.id,
    email: user.email,
    name:  user.displayName,
  })
  window.Klavity?.setMetadata({
    plan:  user.subscriptionTier,
    build: window.__APP_VERSION__,
  })
})

// 3. Subscribe to events
window.Klavity?.on('submit', function(e) {
  console.log('Bug filed:', e.issueKey)
  // myAnalytics.track('Bug filed', { issueKey: e.issueKey })
})

// 4. Open composer from your own UI
document.getElementById('report-btn').onclick = function() {
  window.Klavity?.open('bug')
}
</script>
